SELECT
    WELL_PAID.LAST_NAME AS "LAST_NAME WELL_PAID",
    WELL_PAID.COMMISSION_PCT AS "COMMISSION WELL_PAID",
    BAD_PAID.LAST_NAME AS "LAST_NAME BAD_PAID",
    BAD_PAID.COMMISSION_PCT AS "COMMISSION BAD_PAID"
FROM employees WELL_PAID CROSS JOIN employees BAD_PAID
WHERE WELL_PAID.COMMISSION_PCT >= BAD_PAID.COMMISSION_PCT * 2
    AND BAD_PAID.MANAGER_ID = 100
    AND WELL_PAID.MANAGER_ID = 100
ORDER BY BAD_PAID.LAST_NAME
